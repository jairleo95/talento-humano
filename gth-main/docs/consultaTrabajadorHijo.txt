  SELECT tr.ID_TRABAJADOR,TR.AP_PATERNO,
    TR.AP_MATERNO,
    TR.NO_TRABAJADOR,
    TR.TI_DOC,
    TR.NU_DOC,
    EM.ID_EMPLEADO,
    EM.CO_APS,
    DHT.ID_DATOS_HIJOS_TRABAJADOR ,
    DHT.AP_PATERNO AS AP_PATERNO_HIJO,
    DHT.AP_MATERNO AS AP_MATERNO_HIJO,
    DHT.NO_HIJO_TRABAJADOR,
    DHT.ES_SEXO,
    TO_CHAR(DHT.FE_NACIMIENTO,'dd/mm/yyyy') AS FE_NACIMIENTO_HIJO,
    DHT.ES_TIPO_DOC,
    DHT.NU_DOC AS NU_DOC_HIJO ,
    TR.AP_NOMBRES_C,
    TR.NU_DOC_C,
    TR.ES_TRABAJA_UPEU_C,
    TRUNC((TRUNC((TO_DATE(TO_CHAR(SYSDATE,'dd/mm/yyyy'))-TO_DATE(TO_CHAR(DHT.FE_NACIMIENTO,'dd/mm/yyyy'))),0))/365.25,0)AS EDAD_HIJO
  FROM RHTM_TRABAJADOR TR
  LEFT OUTER JOIN $.post("Usuario EM
  ON EM.ID_TRABAJADOR=TR.ID_TRABAJADOR
  left outer join
    RHTD_DATOS_HIJO_TRABAJADOR DHT
  on DHT.ID_TRABAJADOR=TR.ID_TRABAJADOR  order by DHT.NO_HIJO_TRABAJADOR asc;
